/**
 * @class MealRepo
 * @description Repository class for handling meal-related database operations.
 */
class MealRepo {
    /**
     * @constructor
     * @param {Object} db - Database client instance for executing queries.
     */
    constructor(db) {
        this.db = db;
    }

    /**
     * @method createNewMeal
     * @description Inserts a new meal record with specified fat and sugar levels.
     * @param {number} fat_level - Fat level of the meal.
     * @param {number} sugar_level - Sugar level of the meal.
     * @returns {Object} The newly created meal record.
     * @throws Throws an error if the query fails.
     */
    async createNewMeal(fat_level, sugar_level) {
        try {
            const query = `INSERT INTO meals (fat_level, sugar_level) VALUES ($1, $2) RETURNING *`;
            const result = await this.db.query(query, [fat_level, sugar_level]);
            return result.rows[0];
        } catch (error) {
            console.error("Error creating new meal:", error);
            throw error;
        }
    }

    /**
     * @method getMealByID
     * @description Retrieves a meal record by its unique ID.
     * @param {number} mealId - The ID of the meal to retrieve.
     * @returns {Object|null} The meal record if found, or null if not found.
     * @throws Throws an error if the query fails.
     */
    async getMealByID(mealId) {
        try {
            const query = `SELECT * FROM meals WHERE id = $1`;
            const result = await this.db.query(query, [mealId]);
            return result.rows.length > 0 ? result.rows[0] : null;
        } catch (error) {
            console.error("Error fetching meal by ID:", error);
            throw error;
        }
    }

    /**
     * @method updateMeal
     * @description Updates fat and sugar levels for a specified meal.
     * @param {number} mealId - The ID of the meal to update.
     * @param {number} fat_level - New fat level for the meal.
     * @param {number} sugar_level - New sugar level for the meal.
     * @returns {boolean} True if the meal was updated, false otherwise.
     * @throws Throws an error if the query fails.
     */
    async updateMeal(mealId, fat_level, sugar_level) {
        try {
            const query = `UPDATE meals SET fat_level = $1, sugar_level = $2 WHERE id = $3 RETURNING *`;
            const result = await this.db.query(query, [fat_level, sugar_level, mealId]);
            return result.rows.length > 0;
        } catch (error) {
            console.error("Error updating meal:", error);
            throw error;
        }
    }

    /**
     * @method deleteMeal
     * @description Deletes a meal record by its unique ID.
     * @param {number} mealId - The ID of the meal to delete.
     * @returns {boolean} True if the meal was deleted, false otherwise.
     * @throws Throws an error if the query fails.
     */
    async deleteMeal(mealId) {
        try {
            const query = `DELETE FROM meals WHERE id = $1 RETURNING *`;
            const result = await this.db.query(query, [mealId]);
            return result.rows.length > 0;
        } catch (error) {
            console.error("Error deleting meal:", error);
            throw error;
        }
    }
}

export default MealRepo;
