/**
 * @class DayMealRepo
 * @description Repository class for handling day meal-related database operations.
 */
class DayMealRepo {
    /**
     * @constructor
     * @param {Object} db - Database client instance for executing queries.
     */
    constructor(db) {
        this.db = db;
    }

    /**
     * @method createNewDayMeal
     * @description Inserts a new day meal record for a specified user, day, meal type, and meal.
     * @param {number} userId - The ID of the user.
     * @param {number} dayId - The ID of the day.
     * @param {number} mealTypeId - The ID of the meal type.
     * @param {number} mealId - The ID of the meal.
     * @returns {Object|null} The inserted day meal record, or null if a duplicate entry exists.
     * @throws Throws an error if the query fails and is not due to a duplicate entry.
     */
    async createNewDayMeal(userId, dayId, mealTypeId, mealId) {
        try {
            const query = `INSERT INTO day_meals (fk_user_id, fk_day_id, fk_meal_type_id, fk_meal_id) VALUES ($1, $2, $3, $4) RETURNING *`;
            const result = await this.db.query(query, [userId, dayId, mealTypeId, mealId]);
            return result.rows[0];
        } catch (error) {
            // Check if the error is a duplicate key violation
            if (error.code === '23505') { // PostgreSQL duplicate key error code
                console.warn(`Duplicate entry for userId: ${userId}, dayId: ${dayId}, mealTypeId: ${mealTypeId}`);
                return null;
            } else {
                console.error("Error creating new day meal entry:", error);
                throw error;
            }
        }
    }

    /**
     * @method getDayMeal
     * @description Retrieves a specific day meal record by user ID, day ID, and meal type ID.
     * @param {number} userId - The ID of the user.
     * @param {number} dayId - The ID of the day.
     * @param {number} mealTypeId - The ID of the meal type.
     * @returns {Object|null} The day meal record if found, or null if not found.
     * @throws Throws an error if the query fails.
     */
    async getDayMeal(userId, dayId, mealTypeId) {
        try {
            const query = `SELECT * FROM day_meals WHERE fk_user_id = $1 AND fk_day_id = $2 AND fk_meal_type_id = $3`;
            const result = await this.db.query(query, [userId, dayId, mealTypeId]);
            return result.rows.length > 0 ? result.rows[0] : null;
        } catch (error) {
            console.error("Error fetching day meal:", error);
            throw error;
        }
    }

    /**
     * @method getDayMealsByUserIDAndDayID
     * @description Retrieves all day meal records for a specific user and day.
     * @param {number} userId - The ID of the user.
     * @param {number} dayId - The ID of the day.
     * @returns {Array} Array of day meal records.
     * @throws Throws an error if the query fails.
     */
    async getDayMealsByUserIDAndDayID(userId, dayId) {
        try {
            const query = `SELECT * FROM day_meals WHERE fk_user_id = $1 AND fk_day_id = $2`;
            const result = await this.db.query(query, [userId, dayId]);
            return result.rows;
        } catch (error) {
            console.error("Error fetching day meals by user ID and day ID:", error);
            throw error;
        }
    }
}

export default DayMealRepo;
